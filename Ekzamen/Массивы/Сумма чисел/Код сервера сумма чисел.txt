int GetNumSocket()
{
	LPSOCKET_INFORMATION temp= NULL;
	temp = SocketInfoList;
	int n = 0;
	while(temp)
	{
		n++;
		temp=temp->Next;
	}
	return n;
}

int Check(char* Str)

}
