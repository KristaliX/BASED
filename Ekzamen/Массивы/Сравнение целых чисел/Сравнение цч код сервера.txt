int GetNumSocket()
{
	LPSOCKET_INFORMATION temp= NULL;
	temp = SocketInfoList;
	int n = 0;
	while(temp)
	{
		n++;
		temp=temp->Next;
	}
	return n;
}

int Check(char* Str1, char* Str2)
{
	char temp1[200], temp2[200];
	int num1, num2;
	num1 = atoi(Str1);
	num2 = atoi(Str2);
	itoa(num1, temp1, 10);
	itoa(num2, temp2, 10);
	if ((strcmp(Str1, temp1) != 0) || (strcmp(Str2,temp2) != 0))
		return -2;
	if (num1 > num2)
		return 1;
	else if(num1 == num2)
		return 0;
	else
		return -1;
}
